class LongestIncreasingSubsequence   {
    public static int findLIS(int[] input, int length, int dpList) {
    int max = 0;
    dp = dpList;
    for (int i = 1 ; i < length ; i = i + 1) {
    for (int j = 0 ; j < i ; j = j + 1) {
    if (input[j] > input[i]) {
    if (dp[i] > (dp[j] + 1))
{
    dp[i] = dp[i];
} else {
    dp[i] = dp[j] + 1;
}
    
}
}
}
    for (int i = 0 ; i < length ; i = i + 1) {
    if (dp[i] > max) {
    max = dp[i];
}
}
    return max;
}

    public static void main(String[] args) {
    int[] input = [10, 22, 9, 33, 21, 50];
    int[] dpList = [1, 1, 1, 1, 1, 1];
    int result = findLIS(input, 6, dpList);
    print(result);
}

}
